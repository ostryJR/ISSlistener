	function checkLatLon(oF)
	{	
		var inputBad=false;
		
		var lat=oF.lat.value;
		var lng=oF.lng.value;
		var loc=oF.loc.value;
		if(oF.ele.value=="")
			oF.ele.value=0;
		var ele=parseInt(oF.ele.value);
		
		if(lat=="")
			lat=0;
		if(lng=="")
			lng=0;
			
		if(isNaN(lat))
			lat=0;
		if(isNaN(lng))
			lng=0;

		if(loc != "" && parseFloat(lat)==0 || parseFloat(lng) == 0)
		{
			loc2deg(oF);
			alert("No valid latitude and longitude detected - attempting to compute from gridsquare");
		}	

		var lat=oF.lat.value;
		var lng=oF.lng.value;
		var loc=oF.loc.value;
		
		if(isNaN(parseFloat(lat)) || isNaN(parseFloat(lng)))
			inputBad=true;
		
		if(lng.indexOf(" ") != -1 || lng.indexOf(" ") != -1)
			inputBad=true;
		
		if(lng.indexOf(",") != -1 || lng.indexOf(",") != -1)
			inputBad=true;

			
		if(inputBad==true)
			alert("Your latitude or longitude is not entered correctly.  Lat and Lon must be decimal degrees and must not contain any spaces or punctuation other than than the decimal point (period).");

		if(ele > 8840 || ele < -396 || isNaN(ele))
		{
			inputBad=true;
			alert("Error - Your elevation must not be greater than 8840 meters or less than -396 meters");
		}
	
		return(!inputBad);
	
	}
